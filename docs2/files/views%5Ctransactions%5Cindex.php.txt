<script>
	function confirmarEliminacion($id)
	{
		eliminar = confirm("¿Deseas eliminar este registro?");
		if (eliminar)
			window.location.href = "<?php echo APP_URL.'/transactions/delete/';?>"+$id;
		else
			alert("Eliminación cancelada");
	}
</script>
